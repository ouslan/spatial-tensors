%------------------------------------------------
\section{Introduction}
%------------------------------------------------

\begin{frame}{Why Spatial Dependence Matters}

	\begin{itemize}
		\item Regional data often exhibit spatial dependence.
		\item Example: Puerto Rico municipalities.
		\item Productivity in one region may depend on neighboring regions.
		\item Violates classical i.i.d. assumption.
	\end{itemize}

\end{frame}

%------------------------------------------------
\section{Production Function}
%------------------------------------------------

\begin{frame}{Production Function}

	\[
		\begin{split}
			Y       & = A K^\alpha L^\beta                                                             \\
			\log(Y) & = \log(A) + \alpha \log(K) + \beta \log(L) +\epsilon ; \epsilon \sim N(0,\sigma)
		\end{split}
	\]

	\begin{itemize}
		\item $Y$ is the total output (real GDP) produced.
		\item $A$ is total factor productivity (TFP), capturing the efficiency with which inputs are used.
		\item $K$ is the quantity of physical capital used in production.
		\item $L$ is the quantity of labor employed.
	\end{itemize}

\end{frame}

%------------------------------------------------
\section{Production Function Cont.}
%------------------------------------------------

\begin{frame}{Production Function}

	\[
		\begin{split}
			Y       & = A K^\alpha L^\beta                                                             \\
			\log(Y) & = \log(A) + \alpha \log(K) + \beta \log(L) +\epsilon ; \epsilon \sim N(0,\sigma)
		\end{split}
	\]

	\begin{itemize}
		\item $\alpha$ is the output elasticity of capital
		\item $\beta$ is the output elasticity of labor
		\item $i$ municipalities id
		\item $t$ time period
	\end{itemize}

\end{frame}


%------------------------------------------------
\section{Parametric Spatial Models}
%------------------------------------------------

\begin{frame}{Spatial Autoregressive (SAR) Model}

	\[
		y = \rho Wy + X\beta + \varepsilon
	\]

	\begin{itemize}
		\item $\rho$ measures spatial dependence.
		\item $W$ defines neighborhood structure.
		\item Sensitive to correct specification of $W$.
	\end{itemize}

\end{frame}

%------------------------------------------------
\section{The Spatial Weights Matrix}
%------------------------------------------------

\begin{frame}{Defining $W$: The Core Challenge}

	\begin{itemize}
		\item $W$ is assumed known in theory.
		\item In practice, it must be chosen.
		\item No universally accepted selection rule.
		\item Misspecification leads to biased estimates.
	\end{itemize}

\end{frame}

\section{Semi-Parametric Alternative}
%------------------------------------------------

\begin{frame}{Tensor Product Spatial Smoother}

	\[
		\log(Y_{it}) =
		\alpha + \beta_1 \log(K_{it}) + \beta_2 \log(L_{it})
		+ f(\text{lat}_i,\text{lon}_i) + \varepsilon_{it}
	\]

	\begin{block}{Key Idea}
		No need to specify $W$.
		Spatial effects captured flexibly using splines.
	\end{block}

\end{frame}

	\textbf{Finding:}

\end{frame}

%------------------------------------------------
\section{Conclusions}
%------------------------------------------------

\begin{frame}{Main Conclusions}

	\begin{itemize}
		\item Parametric models are highly sensitive to $W$.
		\item Misspecification leads to large estimation errors.
		\item Semi-parametric tensor approach provides robust spatial control.
		\item Useful in contexts like Puerto Rico where $W$ is uncertain.
	\end{itemize}

\end{frame}
