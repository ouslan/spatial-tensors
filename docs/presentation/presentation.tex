% --- Introduction ---
\begin{frame}{Introduction}
	\begin{itemize}
		\item \textbf{Spatial Modeling:} Often assumes i.i.d. errors, but spatial proximity creates dependencies.
		\item \textbf{The $W$ Problem:} Spatial Durbin Models (SDM) require a Weight Matrix ($W$), but choosing between Rook, Queen, or KNN is often arbitrary.
		\item \textbf{Research Aim:} Compare the performance of predefined $W$ matrices against semi-parametric spatial smoothers (GAMs).
	\end{itemize}
\end{frame}

% --- Introduction ---
\begin{frame}{Spatial Modeling}

	\begin{columns}

			\textbf{Contiguity Types:}
			\begin{itemize}
				\item \textbf{Rook:} Shared boundaries.
				\item \textbf{Queen:} Boundaries + vertices.
				\item \textbf{KNN:} K-Nearest Neighbors based on distance.
			\end{itemize}
		\end{column}
		\begin{column}{0.5\textwidth}

		\end{column}
	\end{columns}
\end{frame}

% --- Mathematical Framework ---
\begin{frame}{Mathematical Framework}
	\textbf{Cobb-Douglas with Spatial Lag (SAR):}
	\begin{equation}
		\log(Y_{it}) = \alpha + \alpha \log(K_{it}) + \beta \log(L_{it}) + \rho \sum_{j=1}^N w_{ij} \log(Y_{jt}) + \epsilon_{it}
	\end{equation}

	\vfill
	\textbf{Semi-Parametric Specification (Tensor Product):}
	\begin{equation}
		\log(Y_{it}) = \alpha + \alpha \log(K_{it}) + \beta \log(L_{it}) + f(\text{lat}_i, \text{lon}_i) + \epsilon_{it}
	\end{equation}
	$f(\text{lat}, \text{lon})$ uses splines to capture spatial effects without needing $W$.
\end{frame}

% --- Tensor Product Visualization ---
\begin{frame}{Semi-Parametric Spatial Smoothers}

	The spatial smoother is constructed via a Kronecker product of basis functions:
	\begin{block}{The Tensor Product}
		\[ X_{\text{tensor}} = B \otimes D \]
		This generates a smooth surface over the geographical domain, allowing productivity to vary across space flexibly.
	\end{block}
\end{frame}

% --- Conclusion ---
\begin{frame}{Success and Impact}
	\begin{itemize}
		\item \textbf{Sensitivity:} Parametric models (SAR/SDM) are highly sensitive to $W$ choice.
		\item \textbf{Robustness:} The semi-parametric approach offers a reliable "starting point" for spatial control.
		\item \textbf{Application:} Provides a clearer picture of municipal productivity in Puerto Rico despite the lack of official GDP data.
	\end{itemize}
\end{frame}
